import tensorflow as tf
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import Conv2D, MaxPooling2D, Flatten, Dense

def create_tiny_cnn(input_shape=(64, 64, 3), num_classes=2):
    """
    Create a Tiny CNN model for anomaly detection.
    Args:
        input_shape: Shape of the input (e.g., 64x64 RGB image)
        num_classes: Number of output classes (binary classification: Normal, Attack)
    Returns:
        model: Compiled CNN model
    """
    model = Sequential([
        Conv2D(32, (3, 3), activation='relu', input_shape=input_shape),
        MaxPooling2D(pool_size=(2, 2)),
        Conv2D(64, (3, 3), activation='relu'),
        MaxPooling2D(pool_size=(2, 2)),
        Flatten(),
        Dense(64, activation='relu'),
        Dense(num_classes, activation='softmax')
    ])
    
    model.compile(optimizer='adam', loss='sparse_categorical_crossentropy', metrics=['accuracy'])
    return model
